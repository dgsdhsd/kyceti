int dele(mm p, mm q) {
	if (head->next == rear->next)return 1;
	q->next = p->next;
	if (head->next == p)
		head->next = p->next;
	if (rear->next == p)
		rear->next = q;
	return 0;
}
int fuzhu(char* ori, int site, char* alter,char*cop) {                
	int i,j,k;
	j = 0;
	for (i = 0; ori[i] != '\0'; i++) {
		if (i!= site) {
			cop[j++] = ori[i];
		}
		else
			for (k = 0; alter[k] != '\0'; k++)
			{
				cop[j++] = alter[k];
			}
	}
	cop[j] = '\0';
	return 0;
}
